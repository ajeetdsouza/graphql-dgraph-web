```graphql
query MyQuery2 {
  allDirectory(filter: {name: {eq: "pages"}}) {
    nodes {
      relativePath
      name
      children {
        id
        ... on Directory {
          id
          name
          children {
            id
          }
        }
      }
    }
  }
}
```

```graphql
{
  allFile(filter: {extension: {eq: "mdx"}}) {
    edges {
      node {
        extension
        dir
        name
        relativeDirectory
      }
    }
  }
}
```
